local ACTIONS_IN_DISGUISE = { "Cue6" }

local function safeGet(base: Instance?, path: string): TextButton?
	if not base then
		return nil
	end
	for _, pathPiece in string.split(path, ".") do
		local child = base:FindFirstChild(pathPiece)
		if not child then
			return nil
		end
		base = child
	end
	return base
end

local function loopOverButtons(base: Instance?, func: (TextButton) -> nil): nil
	if not base then
		return nil
	end
	for _, thing in base:GetChildren() do
		if thing:IsA("ImageButton") or thing:IsA("TextButton") then
			func(thing)
		end
	end
	return
end

			if button:IsA("GuiButton") and not table.find(cores, button.Name) then
				button:Destroy()
			end
		end

			if coreSelectorFrame:FindFirstChild(core) then
				continue
			end
			local button = script.CoreButton:Clone()
			button.Name = core
			button.LayoutOrder = i
			button.GL_ButtonText.Text = core
			button.Parent = coreSelectorFrame
		end
