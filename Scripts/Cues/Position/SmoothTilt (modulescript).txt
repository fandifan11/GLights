local Cue = {}
Cue.IsPhysicsCue = true
Cue.OnDisable = "ResetTilt"

function Cue.Enable(Changer, _Settings, runOnce, cueName, globalCueSettings)
	local fixtures = Changer.Indexer.Motors.Numbered
	local shouldStopRunning = false

	task.spawn(function()
		local degrees = math.rad(Changer.State.Tilt[1] or 0)
		local startDegrees = degrees
		while not shouldStopRunning do
			local firstSpeed = Changer.State.MotorSpeed
			degrees += firstSpeed * 2
			for offset, fixtureIds in fixtures do
				local rad = degrees + math.rad(offset * (180 / #fixtures))
				local tilt = (math.deg(math.cos(rad)) / 1.5) + if globalCueSettings.Overshoot then 0 else 60
				Changer:CueTilt(tilt, fixtureIds)
			end
			task.wait(1 / workspace:GetRealPhysicsFPS())
			if runOnce and degrees - startDegrees > math.pi * 2 then
				shouldStopRunning = true
			end
		end
		Changer:CueDone(cueName)
	end)

	return function()
		shouldStopRunning = true
	end
end

return Cue
