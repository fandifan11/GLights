local Cue = {}
Cue.OnDisable = "ResetTilt"
Cue.IsPhysicsCue = true
Cue._GroupOffsets = {}

local rng = Random.new()

function Cue.Enable(Changer, Settings, shouldRunOnce, cueName, globalCueSettings)
	local shouldStopLooping = shouldRunOnce
	local fixtures = Changer.Indexer.Motors.All
	local numbered = Changer.Indexer.Motors.Numbered
	for group, _ in numbered do
		if not Cue._GroupOffsets[group] then
			Cue._GroupOffsets[group] = rng:NextNumber(0, 1)
		end
	end

	local randomTimeOffsets = {}
	for _, id in fixtures do
		randomTimeOffsets[id] = rng:NextNumber(0, 1)
	end

	local function tilt(lights, degrees)
		if shouldStopLooping then
			return
		end
		Changer:CueTilt(degrees, lights)
	end

	task.spawn(function()
		repeat
			local degrees1 = if globalCueSettings.Overshoot then -70 else 90
			if globalCueSettings.GroupRandom then
				for group, fixtureIds in numbered do
					task.delay(Cue._GroupOffsets[group], tilt, fixtureIds, degrees1)
				end
			else
				for _, fixtureId in fixtures do
					task.delay(randomTimeOffsets[fixtureId], tilt, { fixtureId }, degrees1)
				end
			end
			task.wait(if globalCueSettings.Overshoot then 2.5 else 1.5)
			if shouldStopLooping then
				if globalCueSettings.KillCues then
					return
				else
					break
				end
			end

			local degrees2 = if globalCueSettings.Overshoot then 70 else 25
			if globalCueSettings.GroupRandom then
				for group, fixtureIds in numbered do
					task.delay(Cue._GroupOffsets[group], tilt, fixtureIds, degrees2)
				end
			else
				for _, fixtureId in fixtures do
					task.delay(randomTimeOffsets[fixtureId], tilt, { fixtureId }, degrees2)
				end
			end
			task.wait(if globalCueSettings.Overshoot then 2.5 else 1.5)
			if shouldStopLooping and globalCueSettings.KillCues then
				return
			end
		until shouldStopLooping
		Changer:CueDone(cueName)
	end)

	return function()
		shouldStopLooping = true
	end
end

return Cue
